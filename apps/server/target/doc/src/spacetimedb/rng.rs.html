11impl ReducerContext {
12    /// Generates a random value.
13    ///
14    /// Similar to [`rand::random()`], but using [`StdbRng`] instead.
15    ///
16    /// See also [`ReducerContext::rng()`].
17    pub fn random<T>(&self) -> T
18    where
19        Standard: Distribution<T>,
20    {
21        Standard.sample(&mut self.rng())
22    }

24    /// Retrieve the random number generator for this reducer transaction,
25    /// seeded by the timestamp of the reducer call.
26    ///
27    /// If you only need a single random value, you can use [`ReducerContext::random()`].
28    ///
29    /// # Examples
30    ///
31    /// ```no_run
32    /// # #[cfg(target_arch = "wasm32")] mod demo {
33    /// use spacetimedb::{reducer, ReducerContext};
34    /// use rand::Rng;
35    ///
36    /// #[spacetimedb::reducer]
37    /// fn rng_demo(ctx: &spacetimedb::ReducerContext) {
38    ///     // Can be used in method chaining style:
39    ///     let digit = ctx.rng().gen_range(0..=9);
40    ///
41    ///     // Or, cache locally for reuse:
42    ///     let mut rng = ctx.rng();
43    ///     let floats: Vec<f32> = rng.sample_iter(rand::distributions::Standard).collect();
44    /// }
45    /// # }
46    /// ```
47    ///
48    /// For more information, see [`StdbRng`] and [`rand::Rng`].
49    pub fn rng(&self) -> &StdbRng {
50        self.rng.get_or_init(|| StdbRng::seed_from_ts(self.timestamp))
51    }
52}

99/// A reference to the random number generator for this reducer call.
100///
101/// An instance can be obtained via [`ReducerContext::rng()`]. Import
102/// [`rand::Rng`] in order to access many useful random algorithms.
103///
104/// `StdbRng` uses the same PRNG as `rand`'s [`StdRng`]. Note, however, that
105/// because it is seeded from a publicly-known timestamp, it is not
106/// cryptographically secure.
107///
108/// You may be looking for a level of reproducibility that's finer-grained
109/// than "if it happens at the exact same time, you get the same result" --
110/// if so, just seed an [`StdRng`] directly, or use another rng like those
111/// listed [here](https://rust-random.github.io/book/guide-rngs.html).
112/// Just note that you must not store any state, including an rng, in a global
113/// variable or any other in-WASM side channel. Any and all state persisted
114/// across reducer calls _must_ be stored in the database.
115pub struct StdbRng {
116    // Comments in the rand crate claim RefCell can have an overhead of up to 15%,
117    // and so they use an UnsafeCell instead:
118    // <https://docs.rs/rand/0.8.5/src/rand/rngs/thread.rs.html#20-32>
119    // This is safe as long as no method on `StdRngCell` is re-entrant.
120    rng: UnsafeCell<StdRng>,
121
122    // !Send + !Sync
123    _marker: PhantomData<*mut ()>,
124}
125
126impl StdbRng {
127    /// Seeds a [`StdbRng`] from a timestamp.
128    fn seed_from_ts(timestamp: Timestamp) -> Self {
129        Self {
130            rng: StdRng::seed_from_u64(timestamp.to_micros_since_unix_epoch() as u64).into(),
131            _marker: PhantomData,
132        }
133    }
134}

154impl RngCore for &StdbRng {
155    #[inline(always)]
156    fn next_u32(&mut self) -> u32 {
157        // SAFETY: We must make sure to stop using `rng` before anyone else
158        // creates another mutable reference
159        let rng = unsafe { &mut *self.rng.get() };
160        rng.next_u32()
161    }
